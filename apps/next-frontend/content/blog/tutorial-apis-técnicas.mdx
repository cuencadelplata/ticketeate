    """Verificar la autenticidad de un ticket usando HMAC"""

from google.oauth2 import service_account
from googleapiclient.discovery import build

def sync_to_google_calendar(event_data, calendar_id="primary"):
    """Sincronizar evento con Google Calendar"""
    
    credentials = service_account.Credentials.from_service_account_file(
        'service-account.json',
        scopes=['https://www.googleapis.com/auth/calendar']
    )
    
    service = build('calendar', 'v3', credentials=credentials)
    
    event = {
        'summary': event_data['title'],
        'description': f"{event_data['description']}\n\nTickets: {event_data['ticket_url']}",
        'start': {
            'dateTime': event_data['start_date'],
            'timeZone': event_data['timezone']
        },
        'end': {
            'dateTime': event_data['end_date'],
            'timeZone': event_data['timezone']
        },
        'location': event_data['venue_address'],
        'attendees': [
            {'email': attendee['email']} 
            for attendee in event_data['confirmed_attendees']
        ]
    }
    
    created_event = service.events().insert(
        calendarId=calendar_id, 
        body=event
    ).execute()
    
    return created_event['id']
